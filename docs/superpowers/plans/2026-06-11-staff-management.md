# Staff Management Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the next school-management slice: school admins can add staff, assign principal/teacher roles, view staff, and update staff records for the active school.

**Architecture:** Reuse the existing school actor model. `school_actors` stores the person, `staff_profiles` stores employment details, and `school_actor_roles` stores internal school roles. Keep owner role internal: bootstrap creates owner, staff UI can assign only `principal` and `teacher`.

**Tech Stack:** TanStack Start, Hono/oRPC, Drizzle, Better Auth, Zod, TanStack Query, Paraglide i18n, shadcn-style UI package.

## Permissions

- Any active organization member can list staff.
- Owners and principals can create/update staff.
- Only owners can assign or remove `principal`.
- Owners and principals can assign or remove `teacher`.
- `owner` is not assignable from staff UI or staff API.

  fullName: nonEmptyTextSchema,

  joinedOn: isoDateSchema.nullable(),
  leftOn: isoDateSchema.nullable(),
  phone: z.string().nullable(),
  roles: z.array(staffAssignableRoleSchema),
  status: staffStatusSchema,
  title: z.string().nullable(),
  updatedAt: z.iso.datetime()
});
export type StaffMember = z.infer<typeof staffMemberSchema>;

const staffMemberInputFieldsSchema = z.object({
  department: nonEmptyTextSchema.max(120).nullable().optional(),
  email: z.email().nullable().optional(),
  employeeCode: nonEmptyTextSchema.max(80),
  fullName: nonEmptyTextSchema.max(160),
  joinedOn: isoDateSchema.nullable().optional(),
  leftOn: isoDateSchema.nullable().optional(),
  phone: nonEmptyTextSchema.max(40).nullable().optional(),
  roles: z.array(staffAssignableRoleSchema).min(1).optional(),
  status: staffStatusSchema.optional(),
  title: nonEmptyTextSchema.max(120).nullable().optional()
});

export const staffMemberCreateInputSchema = staffMemberInputFieldsSchema
  .extend({
    roles: z.array(staffAssignableRoleSchema).min(1).default(["teacher"]),
    status: staffStatusSchema.default("active")
  })
  .refine(
    (input) =>
      input.leftOn === undefined ||
      input.joinedOn === undefined ||
      input.leftOn === null ||
      input.joinedOn === null ||
      input.joinedOn <= input.leftOn,
    {
      message: "Joined date must be before or equal to left date.",
      path: ["leftOn"]
    }
  );
export type StaffMemberCreateInput = z.infer<typeof staffMemberCreateInputSchema>;

export const staffMemberUpdateInputSchema = staffMemberInputFieldsSchema
  .partial()
  .extend({
    id: uuidSchema
  })
  .refine(hasUpdateFields, {
    message: "At least one field must be provided.",
    path: ["id"]
  })
  .refine(
    (input) =>
      input.leftOn === undefined ||
      input.joinedOn === undefined ||
      input.leftOn === null ||
      input.joinedOn === null ||
      input.joinedOn <= input.leftOn,
    {
      message: "Joined date must be before or equal to left date.",
      path: ["leftOn"]
    }
  );

export const staffListInputSchema = z.object({});

import {
  type StaffAssignableRole,
  type StaffListOutput,
  type StaffMember,
  type StaffMemberCreateInput,
  type StaffMemberUpdateInput
} from "@tsu-stack/core/school";
import { and, asc, db, eq, inArray } from "@tsu-stack/db";
import {
  member,
  schoolActorRoles,
  schoolActors,
  session,
  staffProfiles
} from "@tsu-stack/db/schema";

function timestampToIso(value: Date | string): string {
  return value instanceof Date ? value.toISOString() : new Date(value).toISOString();
}

type StaffRow = {
  actorEmail: string | null;
  actorFullName: string;
  actorId: string;
  actorPhone: string | null;
  createdAt: Date | string;
  department: string | null;
  employeeCode: string;
  id: string;
  joinedOn: string | null;
  leftOn: string | null;
  organizationId: string;
  status: "active" | "on_leave" | "inactive";
  title: string | null;
  updatedAt: Date | string;
};

function normalizeRoles(roles: string[]): StaffAssignableRole[] {
  return roles.filter(
    (role): role is StaffAssignableRole => role === "principal" || role === "teacher"
  );
}

function staffRowToOutput(row: StaffRow, roles: string[]): StaffMember {
  return {
    actorId: row.actorId,
    createdAt: timestampToIso(row.createdAt),
    department: row.department,
    email: row.actorEmail,
    employeeCode: row.employeeCode,
    fullName: row.actorFullName,
    id: row.id,
    joinedOn: row.joinedOn,
    leftOn: row.leftOn,
    phone: row.actorPhone,
    roles: normalizeRoles(roles),
    status: row.status,
    title: row.title,
    updatedAt: timestampToIso(row.updatedAt)
  };
}

export async function getActiveOrganizationIdForSession(sessionId: string): Promise<string | null> {
  const [row] = await db
    .select({ activeOrganizationId: session.activeOrganizationId })
    .from(session)
    .where(eq(session.id, sessionId))
    .limit(1);

  return row?.activeOrganizationId ?? null;
}

export async function isOrganizationMember(
  organizationId: string,
  userId: string
): Promise<boolean> {
  const rows = await db
    .select({ id: member.id })
    .from(member)
    .where(and(eq(member.organizationId, organizationId), eq(member.userId, userId)))
    .limit(1);

  return rows.length > 0;
}

export async function getSchoolAccessRolesForUser(
  organizationId: string,
  userId: string
): Promise<string[]> {
  const rows = await db
    .select({ role: schoolActorRoles.role })
    .from(schoolActorRoles)
    .innerJoin(
      schoolActors,
      and(
        eq(schoolActors.organizationId, schoolActorRoles.organizationId),
        eq(schoolActors.id, schoolActorRoles.actorId)
      )
    )
    .where(
      and(
        eq(schoolActorRoles.organizationId, organizationId),
        eq(schoolActorRoles.active, true),
        eq(schoolActors.userId, userId)
      )
    );

  return rows.map((row) => row.role);
}

export async function listStaffMembers(organizationId: string): Promise<StaffMember[]> {
  const rows = await db
    .select({
      actorEmail: schoolActors.email,
      actorFullName: schoolActors.fullName,
      actorId: schoolActors.id,
      actorPhone: schoolActors.phone,
      createdAt: staffProfiles.createdAt,
      department: staffProfiles.department,
      employeeCode: staffProfiles.employeeCode,
      id: staffProfiles.id,
      joinedOn: staffProfiles.joinedOn,
      leftOn: staffProfiles.leftOn,
      organizationId: staffProfiles.organizationId,
      status: staffProfiles.status,
      title: staffProfiles.title,
      updatedAt: staffProfiles.updatedAt
    })
    .from(staffProfiles)
    .innerJoin(
      schoolActors,
      and(
        eq(schoolActors.organizationId, staffProfiles.organizationId),
        eq(schoolActors.id, staffProfiles.actorId)
      )
    )
    .where(eq(staffProfiles.organizationId, organizationId))
    .orderBy(asc(schoolActors.fullName), asc(staffProfiles.employeeCode));

  const roleRows = await db
    .select({ actorId: schoolActorRoles.actorId, role: schoolActorRoles.role })
    .from(schoolActorRoles)
    .where(
      and(
        eq(schoolActorRoles.organizationId, organizationId),
        eq(schoolActorRoles.active, true),
        inArray(schoolActorRoles.role, ["principal", "teacher"])
      )
    );

  const rolesByActor = new Map<string, string[]>();
  for (const row of roleRows) {
    rolesByActor.set(row.actorId, [...(rolesByActor.get(row.actorId) ?? []), row.role]);
  }

  return rows.map((row) => staffRowToOutput(row, rolesByActor.get(row.actorId) ?? []));
}

export async function createStaffMember(
  organizationId: string,
  input: StaffMemberCreateInput
): Promise<StaffMember> {
  const [created] = await db.transaction(async (tx) => {
    const [actor] = await tx
      .insert(schoolActors)
      .values({
        email: input.email ?? null,
        fullName: input.fullName,
        organizationId,
        phone: input.phone ?? null,
        status: input.email ? "invited" : "active"
      })
      .returning();

    const [profile] = await tx
      .insert(staffProfiles)
      .values({
        actorId: actor.id,
        department: input.department ?? null,
        employeeCode: input.employeeCode,
        joinedOn: input.joinedOn ?? null,
        leftOn: input.leftOn ?? null,
        organizationId,
        status: input.status,
        title: input.title ?? null
      })
      .returning();

    if (input.roles.length > 0) {
      await tx.insert(schoolActorRoles).values(
        input.roles.map((role) => ({
          actorId: actor.id,
          organizationId,
          role
        }))
      );
    }

    return { actor, profile };
  });

  return staffRowToOutput(
    {
      ...created.profile,
      actorEmail: created.actor.email,
      actorFullName: created.actor.fullName,
      actorId: created.actor.id,
      actorPhone: created.actor.phone
    },
    input.roles
  );
}

export async function updateStaffMember(
  organizationId: string,
  input: StaffMemberUpdateInput
): Promise<StaffMember | null> {
  const { id, roles, ...values } = input;

  const updated = await db.transaction(async (tx) => {
    const [existing] = await tx
      .select({
        actorId: staffProfiles.actorId
      })
      .from(staffProfiles)
      .where(and(eq(staffProfiles.organizationId, organizationId), eq(staffProfiles.id, id)))
      .limit(1);

    if (!existing) {
      return null;
    }

    const actorValues = {
      email: values.email,
      fullName: values.fullName,
      phone: values.phone
    };

    if (Object.values(actorValues).some((value) => value !== undefined)) {
      await tx
        .update(schoolActors)
        .set(actorValues)
        .where(
          and(
            eq(schoolActors.organizationId, organizationId),
            eq(schoolActors.id, existing.actorId)
          )
        );
    }

    await tx
      .update(staffProfiles)
      .set({
        department: values.department,
        employeeCode: values.employeeCode,
        joinedOn: values.joinedOn,
        leftOn: values.leftOn,
        status: values.status,
        title: values.title
      })
      .where(and(eq(staffProfiles.organizationId, organizationId), eq(staffProfiles.id, id)));

    if (roles !== undefined) {
      await tx
        .delete(schoolActorRoles)
        .where(
          and(
            eq(schoolActorRoles.organizationId, organizationId),
            eq(schoolActorRoles.actorId, existing.actorId),
            inArray(schoolActorRoles.role, ["principal", "teacher"])
          )
        );

      if (roles.length > 0) {
        await tx.insert(schoolActorRoles).values(
          roles.map((role) => ({
            actorId: existing.actorId,
            organizationId,
            role
          }))
        );
      }
    }

    return existing.actorId;
  });

  if (!updated) {
    return null;
  }

  return (await listStaffMembers(organizationId)).find((staff) => staff.id === id) ?? null;
}

export function staffPermissionsFromRoles(
  roles: string[]
): Pick<StaffListOutput, "canManagePrincipalRole" | "canManageStaff"> {
  return {
    canManagePrincipalRole: roles.includes("owner"),
    canManageStaff: roles.includes("owner") || roles.includes("principal")
  };
}

import {
  staffListInputSchema,
  staffListOutputSchema,
  staffMemberCreateInputSchema,
  staffMemberSchema,
  staffMemberUpdateInputSchema
} from "@tsu-stack/core/school";

import { type OrpcContext } from "#@/lib/context/types";
import { protectedProcedure } from "#@/lib/procedures/factory";

import {
  createStaffMember,
  getActiveOrganizationIdForSession,
  getSchoolAccessRolesForUser,
  isOrganizationMember,
  listStaffMembers,
  staffPermissionsFromRoles,
  updateStaffMember
} from "./queries";

const schoolStaffProcedure = protectedProcedure.errors({
  ACTIVE_ORGANIZATION_REQUIRED: {
    message: "Select an organization before managing staff.",
    status: 400
  },
  DUPLICATE_STAFF_RECORD: {
    message: "A staff member with the same email or employee code already exists.",
    status: 409
  },
  INVALID_STAFF_DATES: {
    message: "Staff date range is invalid.",
    status: 400
  },
  ORGANIZATION_ACCESS_DENIED: {
    message: "You do not have access to the active organization.",
    status: 403
  },
  SCHOOL_PRINCIPAL_MANAGEMENT_DENIED: {
    message: "Only owners can assign principal access.",
    status: 403
  },
  SCHOOL_STAFF_MANAGEMENT_DENIED: {
    message: "Only owners and principals can manage staff.",
    status: 403
  },
  STAFF_RECORD_NOT_FOUND: {
    message: "Staff member not found.",
    status: 404
  }
});

type AuthenticatedContext = OrpcContext & {
  session: NonNullable<OrpcContext["session"]>;
};

type SchoolStaffErrors = Parameters<
  Parameters<typeof schoolStaffProcedure.handler>[0]
>[0]["errors"];

async function requireActiveOrganization(context: AuthenticatedContext, errors: SchoolStaffErrors) {
  const organizationId = await getActiveOrganizationIdForSession(context.session.session.id);

  if (!organizationId) {
    throw errors.ACTIVE_ORGANIZATION_REQUIRED();
  }

  if (!(await isOrganizationMember(organizationId, context.session.user.id))) {
    throw errors.ORGANIZATION_ACCESS_DENIED();
  }

  return organizationId;
}

async function getStaffPermissions(context: AuthenticatedContext, organizationId: string) {
  const roles = await getSchoolAccessRolesForUser(organizationId, context.session.user.id);
  return staffPermissionsFromRoles(roles);
}

function assertCanManageStaff(permissions: { canManageStaff: boolean }, errors: SchoolStaffErrors) {
  if (!permissions.canManageStaff) {
    throw errors.SCHOOL_STAFF_MANAGEMENT_DENIED();
  }
}

function assertCanManageRequestedRoles(
  roles: readonly string[] | undefined,
  permissions: { canManagePrincipalRole: boolean },
  errors: SchoolStaffErrors
) {
  if (roles?.includes("principal") && !permissions.canManagePrincipalRole) {
    throw errors.SCHOOL_PRINCIPAL_MANAGEMENT_DENIED();
  }
}

function hasDatabaseCode(error: unknown, code: string): boolean {
  if (typeof error !== "object" || error === null) {
    return false;
  }

  if ("code" in error && error.code === code) {
    return true;
  }

  return "cause" in error && hasDatabaseCode(error.cause, code);
}

async function mapStaffDatabaseError<T>(action: () => Promise<T>, errors: SchoolStaffErrors) {
  try {
    return await action();
  } catch (error) {
    if (hasDatabaseCode(error, "23505")) {
      throw errors.DUPLICATE_STAFF_RECORD();
    }

    if (hasDatabaseCode(error, "23514")) {
      throw errors.INVALID_STAFF_DATES();
    }

    throw error;
  }
}

export const schoolStaffRouter = {
  create: schoolStaffProcedure
    .route({
      description: "Create a staff member for the active organization",
      method: "POST"
    })
    .input(staffMemberCreateInputSchema)
    .output(staffMemberSchema)
    .handler(async ({ context, errors, input }) => {
      const organizationId = await requireActiveOrganization(context, errors);
      const permissions = await getStaffPermissions(context, organizationId);
      assertCanManageStaff(permissions, errors);
      assertCanManageRequestedRoles(input.roles, permissions, errors);
      return mapStaffDatabaseError(() => createStaffMember(organizationId, input), errors);
    }),
  list: schoolStaffProcedure
    .route({
      description: "List staff members for the active organization",
      method: "GET"
    })
    .input(staffListInputSchema)
    .output(staffListOutputSchema)
    .handler(async ({ context, errors }) => {
      const organizationId = await requireActiveOrganization(context, errors);
      const permissions = await getStaffPermissions(context, organizationId);
      const staff = await listStaffMembers(organizationId);
      return { ...permissions, staff };
    }),
  update: schoolStaffProcedure
    .route({
      description: "Update a staff member for the active organization",
      method: "PATCH"
    })
    .input(staffMemberUpdateInputSchema)
    .output(staffMemberSchema)
    .handler(async ({ context, errors, input }) => {
      const organizationId = await requireActiveOrganization(context, errors);
      const permissions = await getStaffPermissions(context, organizationId);
      assertCanManageStaff(permissions, errors);
      assertCanManageRequestedRoles(input.roles, permissions, errors);
      const row = await mapStaffDatabaseError(
        () => updateStaffMember(organizationId, input),
        errors
      );

      if (!row) {
        throw errors.STAFF_RECORD_NOT_FOUND();
      }

      return row;
    })
};
